public class Main {
    public static void main ( String[] args ) {
        Scanner sc = new Scanner ( System.in ) ;
        int n ;
        System.out.println ( "n=" ) ;
        n = sc.nextInt ( ) ;
        int i = 0 ;
        int sum = 0 ;
        int x ;
        while ( i < n ) {
            x = sc.nextInt ( ) ;
            sum = sum + x ;
            i = i + 1 ;
        }
        System.out.println ( "sum=" ) ;
        System.out.println ( sum ) ;
    }
}
